# MyBatisPlus

## 常用注解

> @TableName

@TableName 设置表名映射，如果不添加，默认用类名作为表名去映射

404 资源找不到

403 权限不足

400 参数类型错误

500 Java抛异常

MBP 执行查询的时候，会默认将实体类的类名作为表名来查询

Account 类来映射到 user 表

```java
package com.test.entity;

import com.baomidou.mybatisplus.annotation.TableField;
import com.baomidou.mybatisplus.annotation.TableId;
import com.baomidou.mybatisplus.annotation.TableName;
import lombok.Data;

@Data
@TableName("user")
public class Account {
    @TableId("id")
    private Long num;
    @TableField("name")
    private String title;
    private Integer age;
    private String email;
}
```

```java
package com.test;

import com.baomidou.mybatisplus.annotation.TableField;
import com.baomidou.mybatisplus.annotation.TableId;
import com.baomidou.mybatisplus.annotation.TableName;
import com.test.entity.Account;
import java.lang.reflect.Field;

public class Test {
    public static void main(String[] args) {
        //模拟MBP，根据类生成SQL
        Class clazz = Account.class;
        //获取类名
        //如何判断Account是否添加了TableName注解
        TableName annotation = (TableName) clazz.getAnnotation(TableName.class);
        String sql = "select ";
        if(annotation != null){
            //获取表名
            String value = annotation.value();
            //获取属性名
            Field[] declaredFields = clazz.getDeclaredFields();
            for (Field declaredField : declaredFields) {
                TableId annotation1 = declaredField.getAnnotation(TableId.class);
                if(annotation1 == null){
                    TableField annotation2 = declaredField.getAnnotation(TableField.class);
                    if(annotation2 == null){
                        sql += declaredField.getName()+",";
                    } else {
                        String value1 = annotation2.value();
                        sql += value1+",";
                    }
                } else {
                    String value1 = annotation1.value();
                    sql += value1+",";
                }
            }
            sql = sql.substring(0, sql.length()-1);
            sql += " from "+value;
            System.out.println(sql);
        } else {
            String name = clazz.getSimpleName();
            name = name.toLowerCase();
            //select id,name from account
            //获取属性名
            Field[] declaredFields = clazz.getDeclaredFields();
            sql = "select ";
            for (Field declaredField : declaredFields) {
                sql += declaredField.getName()+",";
            }
            sql = sql.substring(0, sql.length()-1);
            sql += " from "+name;
            System.out.println(sql);
        }
    }
}
```

## 常用方法

> selectList()

```java
this.accountMapper.selectList(null);
```

查询全部数据

```java
QueryWrapper<User> queryWrapper = new QueryWrapper<>();
queryWrapper.like("name", "明");
return this.userMapper.selectList(queryWrapper);
```

根据条件查询数据，将条件定义到 QueryWrapper 对象中

> selectBatchIds(Collection)

根据多个id进行批量查询

```java
List<Integer> list = Arrays.asList(2, 3);
return this.userMapper.selectBatchIds(list);
```

> selectById()

根据id查询数据

```java
this.userMapper.selectById(id);
```

> selectByMap(Map)

根据Map查询数据，Map中存入的是查询条件

```java
Map<String,Object> map = new HashMap<>();
map.put("name", "李四");
map.put("age", 20);
return this.userMapper.selectByMap(map);
```

> selectCount()

查询符合条件的记录数

```java
QueryWrapper<User> queryWrapper = new QueryWrapper<>();
queryWrapper.gt("age", 21);
return this.userMapper.selectCount(queryWrapper);
```

> selectOne()

查询单条数据

```java
QueryWrapper<User> queryWrapper = new QueryWrapper<>();
queryWrapper.eq("age", 28);
return this.userMapper.selectOne(queryWrapper);
```

> selectPage()

分页查询

```java
package com.test.entity;

import com.baomidou.mybatisplus.annotation.TableField;
import com.baomidou.mybatisplus.annotation.TableId;
import com.baomidou.mybatisplus.annotation.TableName;
import lombok.Data;

import java.math.BigDecimal;
import java.util.Date;

@Data
@TableName("product_info")
public class Product {
    @TableId("product_id")
    private Integer id;
    @TableField("product_name")
    private String name;
    @TableField("product_price")
    private BigDecimal price;
    @TableField("product_stock")
    private Integer stock;
    @TableField("product_description")
    private String description;
    @TableField("product_icon")
    private String icon;
    @TableField("category_type")
    private Integer type;
    @TableField("create_time")
    private Date createtime;
    @TableField("update_time")
    private Date updatetime;
    @TableField("product_status")
    private Integer status;
}
```

```java
package com.test.controller;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.test.entity.Product;
import com.test.mapper.ProductMapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.List;

@RestController
@RequestMapping("/product")
public class ProductController {

    @Autowired
    private ProductMapper mapper;

    @GetMapping("/page/{page}")
    public Page<Product> page(@PathVariable("page") Integer page){
        Page<Product> productPage = new Page<>(page, 5);
        Page<Product> productPage1 = this.mapper.selectPage(productPage, null);
        return productPage1;
    }
}
```
